import math

def truncate_number(num: float) -> float:
    """Truncate the number to its decimal part.

    Args:
        num (float): A single positive floating point number.

    Returns:
        float: The decimal part of the number.
    
    Raises:
        ValueError: If the input number is not positive.
    
    Examples:
        >>> truncate_number(3.5)
        0.5
        >>> truncate_number(0.0001)
        0.0001
        >>> truncate_number(1e10)
        0.0
    """
    if num < 0:
        raise ValueError("Input must be a positive number.")
    
    integer_part = math.floor(num)
    decimal_part = num - integer_part
    return decimal_part

def truncate_number(number: float) -> float:
    """Extracts and returns the decimal part of a positive floating-point number.

    Args:
        number (float): A positive floating-point number.

    Returns:
        float: The decimal part of the input number, which is always less than 1.

    Raises:
        ValueError: If the input number is not positive.

    Examples:
        >>> truncate_number(3.5)
        0.5
        >>> truncate_number(10.75)
        0.75
        >>> truncate_number(0.1)
        0.1
        >>> truncate_number(1000000.9999)
        0.9999
        >>> truncate_number(5.0)
        0.0
    """

        raise ValueError("Input must be a positive floating-point number.")
    
    decimal_part = number % 1.0
    return decimal_part
