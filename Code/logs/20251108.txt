    """Sorts the book titles based on a custom asc-desc-ending order.

    Args:
        n (int): The number of books.
        books (List[str]): A list of book titles.

    Returns:
        List[int]: A list of indices representing the sorted order of the books.
    """
    
    def custom_sort_key(book: str) -> tuple:
        """Generates a sorting key for the book title based on the specified rules.

        Returns:
            tuple: A tuple representing the sorting key.

import math

def minimum_radius(n: int, k: int, habitats: List[Tuple[int, int]]) -> float:
    """Calculate the minimum radius of a circular park that can encompass at least k bird habitats.

    Args:
        n (int): The number of bird habitats.
        k (int): The number of bird habitats required to be inside the park.
        habitats (List[Tuple[int, int]]): A list of tuples representing the coordinates of the bird habitats.

    Returns:
        float: The minimum radius of the park with at least k bird habitats inside.
    """
