    Separate groups are balanced (each opening brace is properly closed) and may be nested within each other.
    Include any spaces in the input string as part of the returned strings.
    """
    result = []
    stack = []
    start_index = 0

    for i, char in enumerate(paren_string):
        if char == '(':
            if not stack:
                start_index = i
            stack.append(char)
        elif char == ')':
            if stack:
                stack.pop()

    return result
